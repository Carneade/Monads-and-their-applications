\section*{Monads and their applications 9}

\begin{question} 
  Show that the interchange law holds in $\VCAT$.
 \end{question}

\begin{question}
  Let $F \colon \ca{V} \rightarrow \ca{W}$ be a lax monoidal functor. Show that $F$ induces a 2-functor
  \[
 F_{\ast}  \colon  \VCAT \rightarrow \ca{W}\mbox{-}\mathbf{CAT}
  \]
  which sends a $\ca{V}$-category $\ca{A}$ to the $\ca{W}$-category $F_\ast \ca{A}$ with the same objects as $\ca{A}$, and hom-objects between $a$ and $b$ given by $F\bigr(\ca{A}(a,b)\bigl)$. 
\end{question}

\begin{question}
  Let $\Mon(\CAT)$ denote the 2-category of monoidal categories, lax monoidal functors, and monoidal natural transformations. Show that the assignment
  \[
  (-)\mbox{-}\Cat \colon \Mon(\CAT) \rightarrow 2\mbox{-}\CAT
  \]
  which sends $\ca{V}$ to the 2-category $\ca{V}\mbox{-}\Cat$ extends to a 2-functor, with action on 1-cells given by the 2-functor of Exercise~2.
\end{question}


\begin{question}
 Let $T \colon \ca{C} \rightarrow \ca{C}$ be a $\ca{V}$-monad. Complete the proof the $T\mbox{-}\Alg$ has the structure of a $\ca{V}$-category. Hint: at some point in the proof that algebra morphisms compose in the unenriched case, we use associativity, namely when we consider the composite
 \[
 \xymatrix{ TA \ar[r]^{Tf} \ar@{-->}[d] & TB  \ar[d]^{\beta} \ar@{-->}[r] & \cdot \ar@{-->}[d] \\
 \cdot \ar@{-->}[r] & B \ar[r]_{g} & C  }
 \]
  in $\ca{C}$. This switches the operations ``precompose with $\beta$'' to ``postcompose with $\beta$'' and when translating the proof to $\ca{V}$-categories, one has to use the associator at this point.
 \end{question}

\begin{question}%[subtitle=(bonus)]
 Show that $T\mbox{-}\Alg$ has the same universal property in $\VCAT$ as the ordinary category of algebras for an unenriched monad has in $\CAT$. Namely, for each $\ca{V}$-category $\ca{A}$, define the category $T\mbox{-}\mathbf{Act}(\ca{A},\ca{C})$ of $T$-actions (now given by $\ca{V}$-functors with a $\ca{V}$-natural transformation $\rho \colon TG \Rightarrow G$) and show that $U^T \colon T\mbox{-}\Alg \rightarrow \ca{C}$ is the universal $T$-action.
\end{question}
