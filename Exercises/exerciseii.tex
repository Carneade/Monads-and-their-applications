\section*{Monads and their applications 2}

\begin{question} 
  Let $\ca{C}$ be a category with finite coproducts. For an object $c \in \ca{C}$, let $c \slash \ca{C}$ denote the slice category, whose objects are morphisms with domain $c$ and whose morphisms are commutative triangles. Show that the forgetful functor $c \slash \ca{C} \rightarrow \ca{C}$ is monadic (using Beck's theorem) and describe the monad in question.
\end{question}

\begin{question}
 Let $\ca{C}$ be the category of torsion free abelian groups.
 \begin{enumerate}
 \item[(a)] Show that the inclusion $\ca{C} \rightarrow \Ab$ is monadic (using the monadicity theorem).
 
 \item[(b)] Show that the forgetful functor $\Ab \rightarrow \Set$ is monadic.
 
 \item[(c)] Show that the composite $\ca{C} \rightarrow \Set$ of the above two functors is \emph{not} monadic. (Hint: what happens to the canonical presentation of a finite abelian group?)
 \end{enumerate}
\end{question}

\begin{question}
 An object $g \in \ca{C}$ is called a \emph{strong generator} if $\ca{C}(g,-) \colon \ca{C} \rightarrow \Set$ is conservative.
 \begin{enumerate}
 \item[(a)] Assume that $\ca{C}$ has small colimits and that $g$ is a strong generator such that $\ca{C}(g,-)$ preserves small colimits (an object satisfying the latter condition is sometimes called \emph{small projective}). Show that there exists a monoid $M$ such that $\ca{C}$ is equivalent to the category of $M$-sets. (Hint: is $\ca{C}(g,-)$ monadic?)
 
 \item[(b)] Show that the monoid $M$ above is isomorphic to the endomorphism monoid $\ca{C}(g,g)$ of $g$.
 \end{enumerate}
\end{question}

\begin{question}
 Let $F \colon \ca{C} \rightarrow \ca{C}$ be an endofunctor. An \emph{$F$-algebra} is a pair $(c,\gamma)$ of an object $c \in \ca{C}$ and a morphism $\gamma \colon Fc \rightarrow c$ (not subject to any axioms). A morphism of algebras $(c,\gamma) \rightarrow (d,\delta)$ is a morphism $f \colon c \rightarrow d$ making the evident square commutative. We denote the category of $F$-algebras by $F\mbox{-}\Alg$. Show that the forgetful functor $F \mbox{-}\Alg \rightarrow \ca{C}$ is monadic if and only if it has a left adjoint.
\end{question}
